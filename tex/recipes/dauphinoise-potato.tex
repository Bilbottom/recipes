\begin{document}
\recipetitle

\subsection*{Ingredients}

\begin{itemize}[nolistsep]
    \item \ingredient{Potatoes}{500g (thinly sliced)}
    \item \ingredient{Double Cream}{200ml}
    \item \ingredient{Milk}{100ml}
    \item \ingredient{Garlic}{1 clove (crushed)}
    \item \ingredient{Gruyere Cheese}{50g (grated)}
\end{itemize}

\subsection*{Instructions}

\begin{enumerate}
    \item Preheat the oven to 180{\degrees}C.
    \item Place the sliced \blue{potatoes} in a bowl of cold water to prevent them from browning. Rinse then pat dry with kitchen paper.
    \item Place the \blue{cream}, \blue{milk}, and \blue{garlic} in a large saucepan and bring to a gentle simmer. Add the potatoes, cover and simmer for 10 minutes until just tender. Season well.
    \item Transfer to a buttered ovenproof dish and sprinkle over the \blue{cheese}. Bake for 25 minutes until golden.
\end{enumerate}
