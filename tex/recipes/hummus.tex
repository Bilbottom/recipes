\subsection*{Ingredients}

\begin{itemize}[nolistsep]
    \item Lemon Juice --- \quant{1 lemons}
    \item Cumin --- \quant{1/2 teaspoon}
    \item Tahini --- \quant{2 tablespoons}
    \item Chick peas --- \quant{1 cans, \textit{no water}}
    \item Extra Virgin Olive Oil
    \item Salt
    \item Roasted Peppers (from a jar)
\end{itemize}

\subsection*{Instructions}

\begin{enumerate}
    \item Blend the \blue{lemon juice} and \blue{tahini} first. This makes the hummus creamier.
    \item Blend the remainder of the items.
\end{enumerate}
