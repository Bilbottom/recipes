\subsection*{Ingredients}

\textit{This is slightly adjusted with changes from Ravee Long.}

\begin{itemize}[nolistsep]
    \item \ingredient{Water}{2 tablespoons}
    \item \ingredient{Vegetable Oil}{a splash}
    \item \ingredient{Thai Fish Sauce}{2 tablespoons}
    \item \ingredient{Oyster Sauce}{2 tablespoons}
    \item \ingredient{Light Soy Sauce}{2 tablespoons}
    \item \ingredient{Dark Soy Sauce}{2 tablespoons}
    \item \ingredient{Sugar}{3 teaspoons}
    \item \ingredient{Red Birds Eye Chilli}{30g (chopped)}
    \item \ingredient{Mild Red Chilli}{100g-150g (chopped)}
    \item \ingredient{Garlic Cloves}{5-10 (crushed)}
    \item \ingredient{Green Beans}{300g (chopped into short pieces)}
    \item \ingredient{Basil (Thai or regular)}{60g-100g (leaves only)}
    \item \ingredient{Minced Meat (any type)}{500g}
    \item \ingredient{Eggs}{1 per person}
\end{itemize}

Serve dish with plain Jasmine rice.

\subsection*{Instructions}

\begin{enumerate}
    \item Pound the \blue{chillies} and \blue{garlic} into a paste using a pestle and mortar.
    \item Emulsify the \blue{minced meat} with half of the \blue{fish sauce}.
    \item Combine the \blue{water}, and \blue{sugar}, \blue{soy sauces}, \blue{oyster sauce}, and remaining \blue{fish sauce} in a bowl.
    \item Heat a wok on high heat and saute the chilli-garlic paste until the garlic starts to turn golden.
    \item Add the \blue{minced meat} and toss until the meat is no longer in big clumps.
    \item Add the sauce mix and toss until the meat is almost cooked.
    \item Add the \blue{green beans} and toss until the meat is cooked.
    \item Remove from the heat and stir in the \blue{basil}.
    \item Fry the \blue{eggs} in very hot oil until the edges are browned and bubbly.
\end{enumerate}
